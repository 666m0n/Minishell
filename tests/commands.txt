# Tests des redirections

cat < tmp_file

# Tests de pipes simples
ls | wc -l
cat Makefile | grep NAME | wc -l
echo hello | cat -e

# Tests des builtins
pwd | cat
pwd | echo hello
echo hello | pwd

# Tests d'erreurs et commandes invalides
ls | invalidcommand | wc -l
doesnotexist | ls | wc -l

# Tests de pipes avec builtins

env | grep PATH | wc -l

# Tests avec espaces excessifs
ls    |    wc    -l

# Tests de multiples pipes avec grep

# Tests exit code avec commandes invalides

invalidcommand | echo $?
